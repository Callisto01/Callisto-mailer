Simple Send (Service)

Inject the DatabaseMailerService service and call its send() method: 

use Callisto\CallistoMailer\Service\DatabaseMailerService;

class RegistrationController extends AbstractController
{
    public function index(DatabaseMailerService $mailer): Response
    {
        $mailer->send(
            code: 'welcome_user',
            recipient: 'sophie.martin@example.com',
            context: [
                'user' => ['firstName' => 'Sophie', 'lastName' => 'Martin'],
                'loginUrl' => 'https://app.example.com/login'
            ]
        );
        
        return new Response("Email sent successfully!");
    }
}
