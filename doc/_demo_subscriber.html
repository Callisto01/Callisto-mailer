Practical Case 2: EventSubscriber

This subscriber catches successful email deliveries to log audit records in the database. 

<?php

declare(strict_types=1);

namespace App\EventSubscriber;

use App\Entity\EmailLog;
use Callisto\CallistoMailer\Event\AfterTemplateMailSendEvent;
use Doctrine\ORM\EntityManagerInterface;
use Symfony\Component\EventDispatcher\EventSubscriberInterface;

class EmailAuditSubscriber implements EventSubscriberInterface
{
    public function __construct(
        private EntityManagerInterface $em
    ) {}

    public static function getSubscribedEvents(): array
    {
        return [
            AfterTemplateMailSendEvent::class => 'onEmailSent',
        ];
    }

    public function onEmailSent(AfterTemplateMailSendEvent $event): void
    {
        $email = $event->getEmail();
        $code = $event->getCode();

        $recipients = $email->getTo();
        $recipientAddress = count($recipients) > 0 ? $recipients[0]->getAddress() : 'unknown';

        $log = new EmailLog();
        $log->setTemplateCode($code);
        $log->setRecipient($recipientAddress);
        $log->setSubject($email->getSubject());
        $log->setSentAt(new \DateTimeImmutable());

        $this->em->persist($log);
        $this->em->flush();
    }
}
